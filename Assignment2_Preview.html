def scrape_data(category_url):
    # request the category url and convert to a beautiful soup object
    sauce = urllib.request.urlopen(category_url).read()
    soup = bs.BeautifulSoup(sauce,'lxml')
    text = []
    sentiment = []
    stars = []
    # get links for all items that have reviews on the category page and iterate over each link 
    for url in soup.find_all('a'):
        review_url = root_url + url.get('href')
        r_sauce = urllib.request.urlopen(review_url).read()
        r_soup = bs.BeautifulSoup(r_sauce,'lxml')
        # iterate over each review 
        for review in r_soup.find_all('div', class_='review'):
            star = review.find('p',class_='stars').img.get('alt')[0]
            stars.append(star)
            if (int(star)>=4):
                sentiment.append("positive")
            else:
                sentiment.append("negative")
            text.append(review.find('p',class_='text').text)  
    #create a dataframe containing the review text, stars and sentiment
    df = pd.DataFrame({"text":text , "sentiment" : sentiment, "stars":stars})
    return df

Part 1 : Apply Pre-Processing steps to create a numeric representation of the data.

¶

Pre-processing the data involves the following steps: 

1. Clean the review text . This involves removing all non-alphanumeric characters, converting them to lowercase, remove stopwords and stem each word.
2. Convert the text into a bag-of words using a Count-Vectorizer.

We don't have a tf-idf step as experiments including tf-idf approach showed that it does not improve performance significantly.

In [6]:
def preprocess_data(df):
    # Initialize empty array to append clean text 
    corpus = [] 

    # iterate over each review row to clean 
    for i in range(0, df.shape[0]): 

        # remove all non-alphanumeric characters 
        review = re.sub('[^a-zA-Z0-9]', ' ', df['text'][i]) 

        # convert all cases to lower cases 
        review = review.lower() 

        # split to array(default delimiter is " ") 
        review = review.split() 

        # creating PorterStemmer object to take main stem of each word 
        ps = PorterStemmer() 
        # stem all words except the stop-words 
        review = [ps.stem(word) for word in review 
                    if not word in set(stopwords.words('english'))] 

        review = ' '.join(review)

        # append each string to create array of clean text 
        corpus.append(review) 

    # Convert to bag of words usind count vectorizer  
    cv = CountVectorizer(max_features = 1500) 

    # X contains corpus (dependent variable) 
    X = cv.fit_transform(corpus).toarray() 

    # y contains labels of positive or negative reviews 
    y = df.iloc[:, 1].values 
    return X,y

Splitting data for Testing and Training the model¶

We follow the approach of dividing the entire corpus into training and testing data. 25% of the data is kept for testing the model and the remaining 75% is used to train the model. Since the split is done randomly, we are not introducing sampling bias by keeping a seperate testing set.

This approach to splitting the data into seperate testing and training sets was preferred over cross-validation as we had enough data and 75% of the data was still representative of the entire population and gave similar results. We did carry out preliminary tests with cross-validation approach to reach this conclusion. In the training phase of part 2, we print out the cross-validation scores on the the training data, just to verify the same.

def get_test_train_split(X,y):
    # Splitting the dataset into the Training set and Test set 
    X_train, X_test, y_train, y_test = train_test_split(X, y, test_size = 0.25) 
    return X_train, X_test, y_train, y_test

from sklearn.metrics import confusion_matrix,accuracy_score,f1_score, classification_report
def get_performance(y_test,y_pred,verbose=False):
    print("\nPerformance on Test set:")
    print("Confusion Matrix")  
    print(confusion_matrix(y_test, y_pred) )
    print("Accuracy Score:",round(accuracy_score(y_test,y_pred)*100,2))
    print("F1 Score:",round(f1_score(y_test,y_pred,average="weighted")*100,2))
    if(verbose):
        print("\nClassification Report:\n",classification_report(y_test, y_pred))

from sklearn.model_selection import cross_val_score
def get_random_forest_model(X_train, X_test, y_train, y_test,verbose=False):

    model = RandomForestClassifier(n_estimators = 501, 
                                criterion = 'entropy')
    model.fit(X_train, y_train) 
    
    scores = cross_val_score(model,X_train,y_train,cv=5)
    print("\nAccuracy on Training set using cross-validation:",round(scores.mean()*100,2))
    
    # Predicting the Test set results 
    y_pred = model.predict(X_test) 
    get_performance(y_test,y_pred,verbose)
    
    return model, y_pred

def get_naive_bayes_model(X_train, X_test, y_train, y_test,verbose=False):
    
    model = MultinomialNB()
    model.fit(X_train, y_train)
    
    scores = cross_val_score(model,X_train,y_train,cv=5)
    print("\nAccuracy on Training set using cross-validation:",round(scores.mean()*100,2))
    
    # predicting the test results 
    y_pred = model.predict(X_test)
    get_performance(y_test,y_pred,verbose)
    
    return model, y_pred

def get_decision_tree_model(X_train, X_test, y_train, y_test,verbose=False):
    
    dt = DecisionTreeClassifier()
    dt.fit(X_train,y_train)
    
    scores = cross_val_score(dt,X_train,y_train,cv=5)
    print("\nAccuracy on Training set using cross-validation:",round(scores.mean()*100,2))
    
    y_pred = dt.predict(X_test)    
    get_performance(y_test,y_pred,verbose)
    
    return dt, y_pred

def get_knn_model(X_train, X_test, y_train, y_test,verbose=False):
    
    knn = KNeighborsClassifier(n_neighbors=10)
    knn.fit(X_train,y_train)
    
    scores = cross_val_score(knn,X_train,y_train,cv=5)
    print("\nAccuracy on Training set using cross-validation:",round(scores.mean()*100,2))
    
    y_pred = knn.predict(X_test)
    get_performance(y_test,y_pred,verbose)
    
    return knn, y_pred

def get_svm_model(X_train, X_test, y_train, y_test,verbose=False):
    
    svm = SVC(random_state=101, gamma = 'auto')
    svm.fit(X_train,y_train)
    
    scores = cross_val_score(svm,X_train,y_train,cv=5)
    print("\nAccuracy on Training set using cross-validation:",round(scores.mean()*100,2))
    
    y_pred = svm.predict(X_test)
    get_performance(y_test,y_pred,verbose)
    
    return svm, y_pred
